CREATE TABLE `batch_db` (
  `BATCH_CODE` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `BATCH_NAME` text NOT NULL,
  `REG_ID` int(10) unsigned NOT NULL,
  PRIMARY KEY (`BATCH_CODE`) USING BTREE,
  UNIQUE KEY `UNI_REG_ID` (`BATCH_CODE`),
  KEY `REG_ID_FK` (`REG_ID`),
  CONSTRAINT `REG_ID_FK1` FOREIGN KEY (`REG_ID`) REFERENCES `regtable` (`REG_ID`) ON DELETE CASCADE
)ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8
