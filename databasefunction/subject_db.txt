USE CLASSDB;
DROP TABLE IF EXISTS SUBJECT_DB; 
CREATE TABLE `SUBJECT_DB` (
  `SUBJECT_CODE` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `SUBJECT_NAME` text NOT NULL,
  `REG_ID` int(10) unsigned NOT NULL,
  PRIMARY KEY (`SUBJECT_CODE`) USING BTREE,
  UNIQUE KEY `UNI_REG_ID` (`SUBJECT_CODE`)
) ENGINE=InnoDB
